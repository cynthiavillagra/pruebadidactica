# ---------------------------------------------------------------------------
# CORE: Framework Web
# ---------------------------------------------------------------------------

# Flask: Micro-framework web minimalista
# POR QUE FLASK: Didactico, minima magia, ideal para ensenar HTTP
# POR QUE NO DJANGO: Demasiado opinionado para un CRUD simple
# POR QUE NO FASTAPI: Requiere async, anade complejidad innecesaria
Flask>=3.0.0

# Werkzeug: Toolkit WSGI (viene con Flask, pero lo fijamos)
# POR QUE FIJAR: Evitar conflictos de versiones
Werkzeug>=3.0.0


# ---------------------------------------------------------------------------
# CONFIGURACION: Variables de Entorno
# ---------------------------------------------------------------------------

# python-dotenv: Carga variables desde .env
# POR QUE: Separar configuracion del codigo (12-factor app)
# POR QUE NO os.environ DIRECTO: .env es mas comodo en desarrollo
python-dotenv>=1.0.0


# ---------------------------------------------------------------------------

# POR QUE: Integracion nativa con Supabase Auth y RLS
# INCLUYE: postgrest-py, gotrue, realtime
supabase>=2.0.0


# ---------------------------------------------------------------------------
# SEGURIDAD: JWT y Autenticacion

# POR QUE: Validar tokens de Supabase en el backend
# POR QUE NO jose: PyJWT es mas simple y suficiente
PyJWT>=2.8.0


# ---------------------------------------------------------------------------
# UTILIDADES

python-dateutil>=2.8.0


# ---------------------------------------------------------------------------
# TESTING (Desarrollo)

# pytest: Framework de testing

# pytest-cov: Cobertura de codigo
# POR QUE: Medir que porcentaje del codigo esta testeado
pytest-cov>=4.0.0


# ---------------------------------------------------------------------------

httpx>=0.25.0
